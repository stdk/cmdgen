!set module=dhcpv6_server prefix=xml\commands\dhcpv6\dhcpv6_server\ feature=DHCPv6_Server

$pool_name=<pool_name 12>;

!context config (config)
enable dhcpv6_server
disable dhcpv6_server
create dhcpv6 pool $pool_name
delete dhcpv6 pool [$pool_name | all]
config dhcpv6 pool ipv6network_addr $pool_name [begin <begin_address IpV6address>] [end <end_address IpV6address>] [prefix <prefix 0-128>]
config dhcpv6 pool domain_name $pool_name <domain_name 240>
config dhcpv6 pool dns_server $pool_name <dns1 IpV6address> {<dns2 IpV6address>}
config dhcpv6 pool lifetime $pool_name [preferred_lifetime <preferred 60-4294967295>] [valid_lifetime <valid 60-4294967295>]

$params = [begin <begin_address IpV6address>] [end <end_address IpV6address>];
config dhcpv6 pool excluded_address $pool_name [add $params | delete [$params|all]]

config dhcpv6 pool manual_binding $pool_name [[add <address IpV6address>] [client_duid <duid 28>] | delete [<address IpV6address> | all]]

$state_enum=[enable|disable];
config dhcpv6_server state $state_enum [[ipif <ipif ciscoPortChannelVlan>]|all]


!context EXEC
show dhcpv6 pool {$pool_name}
show dhcpv6 manual_binding {$pool_name}
show dhcpv6 binding {$pool_name}
clear dhcpv6 binding {$pool_name}
show dhcpv6_server {ipif <ipif ciscoPortChannelVlan>}
show dhcpv6 excluded_address {$pool_name}

!xml out
